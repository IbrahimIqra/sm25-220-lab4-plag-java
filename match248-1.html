 private int hashFunction( Integer key ){
      if(key == 0){
        return -1;
      }
      return (key + 3)% ht.length;
    }

public void remove( Integer key ){


        int index = hashFunction(key);
        PairNode head = ht[index];
        if(head == null){
          return;
        }
        if(head.key.equals(key)){
          ht[index] = head.next;
          return;
        }


        PairNode prev = head;
        PairNode curr = head.next;
        while(curr != null){
          if(curr.key.equals(key)){
            prev.next = curr.next;
            return;
          }
          prev = curr;
          curr = curr.next;
        }
    }

public static String removeConsecDups(String word){
      if(word == null){
        return word;
      }


      LinkedListQueue queue = new LinkedListQueue();
      for(int i = 0; i< word.length(); i++){
        queue.enqueue(word.charAt(i));
      }
      String result = "";
      String prev = (String)queue.dequeue();
      result += prev;
      while(! queue.isEmpty()){
        String curr = (String)queue.dequeue();
        if(curr != prev){
          result += curr;
          prev = curr;
        }
      }
      return result;
    }

 public static int diamondCount(Stack stack, String str) {
        int count = 0;
        for(int i = 0; i < str.length(); i++){
          char ch = str.charAt(i);
          if(ch == '<'){
            stack.push(ch);
          }
          else if(ch == '>'){
            if(! stack.isEmpty() && ((Character)stack.peek())=='<'){
              stack.pop();
              count++;
            }
          }
        }
        return count; // Delete this line
