 public static int diamondCount(Stack stack, String str) {
       int AK=0;//for counting the number of diamond
       for(int AV=0;AV<str.length();AV++){//for loop to iterate every single character of the string str
         char AG =str.charAt(AV);//storing the AV number character into AG 
         if(AG=='>'){//if condition to check '>" 
           if(!stack.isEmpty()){//2nd condition to check if the stack is empty
             if((char)stack.peek()=='<'){//3rd condition to check if the top is '<'
               stack.pop();//remove '<'
               AK++;//increment count
             }
           }
         }
         else if(AG=='<'){// AG is '<'
             stack.push(AG);// then push it into stack so that later can be matched with '>'
           }
         else{}
        
         }
        return AK;//return count
    }

public static String removeConsecDups(String word){
      LinkedListQueue AD= new LinkedListQueue();// created new object
      String AF="";// a variable for storing new string
      for(int AG=0; AG<word.length();AG++){// for loop to iterate thorugh every string character
        AD.enqueue(word.charAt(AG));// adding the character to the queue
      }
      if(!AD.isEmpty()){//checking it the stack is empty
        char AI=(char)AD.dequeue();//1st character out/dequeue
        AF+=AI;//add it to string
        while(!AD.isEmpty()){//while loop for iterating rest characters
          char AJ=(char)AD.dequeue();//2nd character out
          if(AJ==AI){//checking weather 2nd and 1st characters are similar 
          }
          else{//if not
            AF+=AJ;//storing 2nd character to AF
            AI=AJ;//2nd/current character becomes 1st character
          }
        }
      }
      return AF;//return new string
    }
