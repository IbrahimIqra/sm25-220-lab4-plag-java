public class StackAssgnTaskTester {



    public static int diamondCount(Stack stack, String str) {
        int dmd_cnt = 0;

        for (int i = 0; i < str.length(); i++) {
            char d = str.charAt(i);

            if (d == '<') {
                stack.push(d);
            } 
            else if (d == '>') {
                if (!stack.isEmpty() && (char) stack.peek() == '<') {
                    stack.pop(); 
                    dmd_cnt++;
                }
            }
        }

        return dmd_cnt;
    }
    }

public class RemoveConsecutiveDuplicates {



    public static String removeConsecDups(String word) {
        LinkedListQueue temp_que = new LinkedListQueue();

        
        for (int i = 0; i < word.length(); i++) {
            temp_que.enqueue(word.charAt(i));
        }

        String ans = "";

        if (!temp_que.isEmpty()) {
            char prev = (char) temp_que.dequeue(); 
            ans += prev; 
            
            while (!temp_que.isEmpty()) {
                char current = (char) temp_que.dequeue();
                if (current != prev) {
                    ans += current;
                    prev = current;
                }
            }
        }

        return ans;
    }

   
}
