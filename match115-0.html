public void remove(Integer key ){
        int i = hashFunction(key);
        PairNode current = ht[i];
        PairNode previous = null;

        while (current != null) {
            if(current.key.equals(key)){
                if(previous == null){
                    ht[i] = current.next;
                }
                else{
                    previous.next = current.next;
                }
                return;
            }
            previous = current;
            current = current.next;
        }
    }

public static int diamondCount(Stack stack, String str) {
        int total = 0;

        for(int i = 0; i < str.length(); i ++){
            char ch = str.charAt(i);

            if(ch == '<'){
                stack.push(ch);
            }
            else if(ch == '>'){
                if(!stack.isEmpty() && (char)stack.peek() == '<'){
                    stack.pop();
                    total++;
                }
            }
        }
        return total;
    }

public static String removeConsecDups(String word){ 
        if(word.length() == 0){


            return "";
        }
        LinkedListQueue q = new LinkedListQueue();

        for(int i = 0; i < word.length(); i++){
            q.enqueue(word.charAt(i));
        }

        char previous = (char)q.dequeue();


        String result = "" + previous;

        while(!q.isEmpty()){
            char current = (char) q.dequeue();
            if(current != previous){
                result += current;
                previous = current;
            }
        }
	    return result;
    }
