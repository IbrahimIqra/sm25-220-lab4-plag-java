   private int hashFunction( Integer key ){
      if (key==null){ 
        return -1;
      }


      return (key+3)%ht.length;
    }  
      


   public void remove(Integer key){
    int k = hashFunction(key); 
    PairNode curr = ht[k];

    if (curr == null) {
        return;
    }

    if (curr.key.equals(key)) {
        ht[k] = curr.next;
        return;
    }

    


    while (curr.next != null) {
        if (curr.next.key.equals(key)) {
            curr.next = curr.next.next;
            return;
        }
        curr = curr.next;
    }
}

    public static int diamondCount(Stack stack, String str) {
    int diamonds = 0; 
    for (int i = 0; i < str.length(); i++) {
        char ch = str.charAt(i);

        if (ch == '<') {
            stack.push(ch);
        } else if (ch == '>') {
            if (!stack.isEmpty()) {
                Object topElem = stack.peek();
                if (topElem.equals('<')) {
                    stack.pop();
                    diamonds++;
                }
            }
        }
    }

    return diamonds;
}

 public static String removeConsecDups(String word) {
    LinkedListQueue q = new LinkedListQueue();
    
    for (int i = 0; i < word.length(); i++) {
        q.enqueue(word.charAt(i));
    }
  
    String res = "";    
    char last = 0;     
    int first = 1;      
    
    
    while (!q.isEmpty()) {
        char letter = (char) q.dequeue();   
        
        if (first == 1) {      
            res = res + letter;
            last = letter;
            first = 0;
        } else {
            if (letter != last) {
                res = res + letter;
                last = letter;
            }
        }
    }
    
    return res;
}
