    private int hashFunction( Integer key ){
        int ind=(key+3)%ht.length;


        return ind;
    }

    public void remove( Integer key ){
      int ind=hashFunction(key);
      PairNode curr=ht[ind];
      PairNode prev=null;
      while(curr!=null){
        if(curr.key.equals(key)){
          if(prev==null){
            ht[ind] = curr.next;
          }
          else{
            prev.next=curr.next;
            curr=null;
          }
          break;
        }
        prev=curr;
        curr=curr.next;
      }
    }

    public static int diamondCount(Stack stack,String str) {
      int dim=0;
      for(int i=0;i<str.length();i++){
        char ch=str.charAt(i);
        if(ch=='<'){
          stack.push('<');
        }
        else if(ch=='>'){
          if(!stack.isEmpty()){
            stack.pop();
              dim++;
          }
        }
      }
      return dim;
    }

    public static String removeConsecDups(String word){
      LinkedListQueue q=new LinkedListQueue();    
      for(int i=0;i<word.length();i++){
        q.enqueue(word.charAt(i));
      }
      String result="";
      char prev='';


      while(!q.isEmpty()){
        char curr=(char)q.dequeue();
        if(curr!=prev){
            result+=curr;
            prev=curr;
        }
      }
      return result;
    }
