public static String removeConsecDups(String word){

	// You MUST create a LinkedListQueue Object to solve this task
	LinkedListQueue l1 = new LinkedListQueue();
    for(int i = 0; i < word.length(); i++){
        l1.enqueue(word.charAt(i));
    }
    String no_dup = "";
    if(l1.isEmpty() == false){
        char before = (char)l1.dequeue();
        no_dup = no_dup + before;


        while(l1.isEmpty() == false){
            char curr = (char)l1.dequeue();
        if(curr != before){
            no_dup = no_dup + curr;
            before = curr;
        }
        }
    }
    return no_dup;
