//remove method
public void remove(Integer key){
    int idx= (key + 3) % 6;
    PairNode crt = ht[idx];
    if(crt == null){
    return;
    }


    if(crt.key.equals(key)){
        ht[idx] = crt.next;
        return;
    }
    while(crt.next != null){
        if(crt.next.key.equals(key)){
            crt.next = crt.next.next;
            return;
        }
        crt = crt.next;
    }
}

//diamondCount method


public static int diamondCount(Stack stack, String str){
    int count = 0;
    for (int i = 0; i < str.length(); i++){
        char c = str.charAt(i);
        if (c == '<'){
            stack.push('<');
        } 
        else if (c == '>'){
            if (!stack.isEmpty()){
                char up = (char) stack.peek();
                if (up == '<'){
                    stack.pop();
                    count = count + 1;
                }
            }
        }
    }   
    return count;
}

//removeConsecDups method


public static String removeConsecDups(String word){
    LinkedListQueue que= new LinkedListQueue();
    for (int i = 0; i < word.length(); i++){
        que.enqueue(word.charAt(i));
    }
    String result = "";
    if (!que.isEmpty()){
        char prv = (char)que.dequeue(); 
        result += prv;
        while (!que.isEmpty()){
            char crt = (char) que.dequeue();
            if (crt != prv){
                result += crt;
                prv = crt;
            }
        }
    }
    return result;
}
