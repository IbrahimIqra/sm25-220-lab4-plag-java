private int hashFunction(Integer key){
        int x=(key+3)%ht.length;
        return x;
    }
    public void remove(Integer key){
        int num=hashFunction(key);
        PairNode hd=ht[num];
        if(hd==null){
            return;
        }
        if(hd.key.equals(key)){
            ht[num]=hd.next;
            return;
        }
        PairNode prv=hd;
        PairNode cur=hd.next;
        while(cur!=null){
            if(cur.key.equals(key)){


                prv.next=cur.next;
                return;
            }
            prv=cur;
            cur=cur.next;
        }	
    }

public static int diamondCount(Stack stack, String str){
        int dmd=0;
        for(int i=0;i<str.length();i++){
            char cr=str.charAt(i);
            if(cr=='<'){
                stack.push(cr);
            }
            else if(cr=='>'){
                if(!stack.isEmpty()){
                    if((char)stack.peek()=='<'){
                        stack.pop();
                        dmd++;
                    }
                }
            }
        }
        return dmd;
    }

public static String removeConsecDups(String word){
        LinkedListQueue que=new LinkedListQueue();
        String str="";
        for(int i=0;i<word.length();i++){
            que.enqueue(word.charAt(i));
        }
        if(!que.isEmpty()){


            char prv=(char)que.dequeue();
            str+=prv;
            while(!que.isEmpty()){
                char cur=(char)que.dequeue();
                if(cur!=prv){
                    str+=cur;
                    prv=cur;
                }
            }
            return str;
        }
        else{
            return "";
        }
    }
