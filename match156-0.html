  private int hashFunction( Integer key ){    
    return (key+3)%6;
  }
  public void remove( Integer key ){


    int indx = hashFunction(key);
    PairNode curr = ht[indx];
    PairNode prev = null;
    while(curr!=null){
      if(curr.key.equals(key)){
        if(prev==null){
          ht[indx]=curr.next;
        }
        else{
          prev.next=curr.next;
        }
        return;
      }
      prev=curr;
      curr=curr.next;
    }
  }

public static int diamondCount(Stack stack, String str) {
    int count = 0;
    for (int i=0;i<str.length();i++){
      char ch =str.charAt(i);
      if(ch=='<'){
        stack.push(ch);
      }
      else if(ch=='>'){
        if(!stack.isEmpty() && ((char)stack.peek()=='<')){
          stack.pop();
          count++;
        }
      }
    }
    return count;
  }

public static String removeConsecDups(String word){
    LinkedListQueue queue = new LinkedListQueue();
    if (word.length() == 0){ 


      return "";
    }
    char prevChar = word.charAt(0);
    queue.enqueue(prevChar);
    for (int i = 1; i < word.length(); i++) {
      char currChar = word.charAt(i);
      if (currChar != prevChar) {
        queue.enqueue(currChar);
        prevChar = currChar;
      }
    }
    String result = "";
    while (!queue.isEmpty()) {
      result += queue.dequeue();
    }
    return result;
  }
