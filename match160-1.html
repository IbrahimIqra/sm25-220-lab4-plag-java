private int hashFunction( Integer key ){
    	return (key + 3) % ht.length;
    }
public void remove( Integer key ){
        int index=hashFunction(key);   
        if(ht[index].key==key) {
        	ht[index]=ht[index].next;
        	return;
        }
        PairNode curr=ht[index];
        while(curr.next!=null) {
        	if(curr.next.key==key) {
        		curr.next=curr.next.next;
        		return;
        	}
        	curr=curr.next;       	
        }
    }

public static int diamondCount(Stack stack, String str) {
        int diamond=0;


        for(int i=0;i<str.length();i++) {
        	char a=str.charAt(i);
        	if(a=='<') {
        		stack.push(a);
        	}
        	if(a=='>') {
        		if(!stack.isEmpty()) {
        			stack.pop();
        			diamond++;
        		}
        	}
        }
        return diamond;
    }

public static String removeConsecDups(String word){
    	LinkedListQueue p=new LinkedListQueue();
    	char prev=' ';


    	for(int i=0;i<word.length();i++) {
    		char cur=word.charAt(i);
    		if (i==0 || cur!=prev) {
    	            p.enqueue(cur);
    	            prev=cur;
    	    }    	
    	}
    	String q="";
    	while(!p.isEmpty()) {
    		q=q+p.dequeue();
    	}
    	return q;
    }
