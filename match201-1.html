public static String removeConsecDups(String word) {

for (int i = 0; i < word.length(); i++) {
queue.enqueue(word.charAt(i));
}String answer ="";
Character lastAdded = null;
while (!queue.isEmpty()) {
   Character cur = (Character) queue.dequeue();
     if (lastAdded == null || !lastAdded.equals(cur)) {

        }
    }
