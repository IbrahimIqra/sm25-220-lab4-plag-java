//Task 01
public class HashTable {
  private int hashFunction( Integer key ){
    int htkey = (key+3)%ht.length;


    return htkey;
  }
  public void remove( Integer key ){
    int hashIndex = hashFunction(key);
    PairNode oldNode = ht[hashIndex];
    PairNode newNode = null;
    while(oldNode!=null){
      if(oldNode.key.equals(key)){
        if(newNode==null){
          ht[hashIndex] = oldNode.next;
        }
        else{
          PairNode nextNode = oldNode.next;
          newNode.next = nextNode;
        }
        return;
      }
      newNode = oldNode;
      oldNode = oldNode.next;
    }
  }
}

public class StackAssgnTaskTester {


  public static int diamondCount(Stack stack, String str) {
    int diamondCount = 0;
    for(int i = 0;i<str.length();i++){
      char oldChar = str.charAt(i);
      if(oldChar=='<'){
        stack.push(oldChar);
      }
      else if(oldChar=='>'){
        if(!stack.isEmpty()){
          char newChar = (char)stack.pop();
          if(newChar=='<'&&oldChar=='>'){
            diamondCount++;
          }
        }
      }
    }
    return diamondCount;
  }
}

public class RemoveConsecutiveDuplicates{


  public static String removeConsecDups(String word){
    LinkedListQueue string = new LinkedListQueue();
    for(int i = 0; i<word.length();i++){
      string.enqueue(word.charAt(i));
    }
    String res = "";
    if(!string.isEmpty()){
      char oldChar = (char)string.dequeue();
      res = res + oldChar;
      while(!string.isEmpty()){
        char newChar = (char)string.dequeue();
        if(newChar!=oldChar){
          res = res + newChar;
          oldChar = newChar;
        }
      }
    }
    return res;
  }
}
