    public static int diamondCount(Stack stack, String sr) {
        int partie2 = 0;
        int paato1 = 0;
        for(int i = 0; i<sr.length(); i++ ) {
            if(sr.charAt(i) == '<') {
                paato1++;
            }
            else if(sr.charAt(i) == '>') partie2++;
            
            else {
                stack.pop();
            }
        }
        if(paato1<partie2) {
            return paato1;
        }
        else  {
            return  partie2;
          }
    }

    public static String removeConsecDups(String w){
        // notun linked list Queue object is banano-ed
        LinkedListQueue kyu = new LinkedListQueue();
        for (int i = 0; i < w.length(); i++) {
            kyu.enqueue(w.charAt(i) );
        }
        String kekka = "";
         if (!kyu.isEmpty()) {
            char agerta= (char) kyu.dequeue();
            kekka += agerta ;



            while (!kyu.isEmpty()) {
                char ekn = (char) kyu.dequeue();
                if (ekn != agerta) kekka += ekn;
                // updation
                agerta =ekn;
            }}
       return kekka;
  }
