 public static int diamondCount(Stack stack, String str) {
      int count=0;
      
      for(int i=0; i<str.length();i++){
        char c=str.charAt(i);
        
        if(c=='<'){
          stack.push(c);
        }
        
        if(c=='>'){
          if(!stack.isEmpty()){
          char top=(Character)stack.peek();
          if(top=='<'){
            count++;
            stack.pop();
          }
          }
        }
      }
      return count;
    }

public static String removeConsecDups(String word){


    LinkedListQueue queue=new LinkedListQueue();
    
    for(int i=0;i<word.length();i++){
      queue.enqueue(word.charAt(i));
    }
    String result="";
    Character prev=null;
    
    while(!queue.isEmpty()){
      Character current=(Character)queue.dequeue();
      if(prev==null||current!=prev){
        result+=current;
        prev=current;
      }
    }
    return result;
    }
