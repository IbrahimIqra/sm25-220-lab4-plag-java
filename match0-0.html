private int hashFunction( Integer key ){


    return (key + 3) % ht.length; 
}

public void remove( Integer key ){
    int hash = hashFunction(key);
    if(ht[hash] != null){
        PairNode current;
        PairNode prev = null;
        for(current = ht[hash]; current != null; current = current.next){
            if(current.key == key){
                if(prev == null){
                    ht[hash] = current.next;
                }
                else {
                    prev.next = current.next;
                }
                break;
            }
            prev = current;
        }
    }
}

public static int diamondCount(Stack stack, String str) {
    int count = 0;
    // iterate the string
    for(int i=0; i< str.length(); i++){
        if(str.charAt(i) == '.'){
            continue;
        }
        else if(str.charAt(i) == '<'){
            stack.push("<");
        }
        else if(str.charAt(i) == '>' && stack.peek() != null){
            String last_elem = stack.peek().toString();
            if(last_elem.equals("<")){
                count++;
                stack.pop();
            }
        }
    }

    return count; 
}

public static String removeConsecDups(String word){
    LinkedListQueue queue = new LinkedListQueue();
    String new_string = "";
    for(int i=0; i< word.length(); i++){
        if(queue.isEmpty()){
            queue.enqueue(word.charAt(i));
        }
        else if((char)queue.peek() == word.charAt(i)){
            continue;
        }
        else {
            new_string += queue.dequeue().toString();
            queue.enqueue(word.charAt(i));
        }
    }
    new_string += queue.dequeue().toString();
    return new_string; 
}
