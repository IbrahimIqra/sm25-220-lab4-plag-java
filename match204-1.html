       if(head==null){
        return;
       }
       if(head.key.equals(key)){
        ht[index]=head.next;
        return;
       }
      PairNode prev=head;
       PairNode current=head.next;
       while(current!=null){
        if(current.key.equals(key)){
            prev.next=current.next;
            return;
        }
        prev=current;
        current=current.next;
       }
    }
