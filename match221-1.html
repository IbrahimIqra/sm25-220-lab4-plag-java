private int hashFunction( Integer key ){
        
        return (key + 3) % ht.length ; 
        
    }


public void remove( Integer key ){
        
      int idx = hashFunction (key);
      
      PairNode temp = ht[idx];
      
      PairNode prev = null; 
      
      if (temp == null){
        
        return;
        
      }
      
      while (temp != null){
        
        if(temp.key.equals(key)){
          
          if (prev == null){
            
            ht[idx] = temp.next;
            
            break;
                      
          }
          
          else {
            
            prev.next = temp.next;
            
            break;
           
          }
          
        }
        
        else {
          


          prev = temp;
          
          temp = temp.next;
          
        }
        
      }
          
    }

public static int diamondCount(Stack stack, String str) {
        
      int count = 0;
      
      for ( int i = 0; i < str.length(); i++ ){
        
        char ch = str.charAt(i);
        
        if ( ch == '<' ){
          
          stack.push(ch);
          
        }
        
        else if ( ch == '>' ){
          
          if (!(stack.isEmpty())){
            
            if ((char)stack.peek() == '<'){
              
              stack.pop();
              
              count++;
              
            }
            
          }
          
        }
        
      }
      
      return count;
        
    }

public static String removeConsecDups(String word){
 
      LinkedListQueue newQueue = new LinkedListQueue();
      
      String newWord = "";
      
      char next = ' ';
      
      for (int i = 0; i < word.length(); i++) {
        
        char ch = word.charAt(i);
        


        newQueue.enqueue(ch);
        
      }
      
      while (!(newQueue.isEmpty())){
        
        char temp = (char)newQueue.dequeue();
        
        if (temp != next){
          
          newWord += temp;
          
          next = temp;
          
        }
        
      }
      
      return newWord;
      
    }
