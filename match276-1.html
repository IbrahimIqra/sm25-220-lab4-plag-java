public static String removeConsecDups(String word){
	  if(word == null ){
        return word;

      }



      LinkedListQueue ex = new LinkedListQueue();

      for(int i = 0; i < word.length(); i++){
        ex.enqueue(word.charAt(i));
      }

      String result = "";

      Character esp = null;
      while(!ex.isEmpty()){
        char temp =((Character)ex.dequeue());
        if(esp == null || temp != esp){
            result += temp;
            esp = temp;

        }
      }
	
	return result; 
    }

private int hashFunction( Integer key ){
        return (key + 3) % ht.length;

       
    }


 public void remove( Integer key ){
        int find = hashFunction(key);

        PairNode temp = ht[find];

        PairNode previous = null;

        while(temp != null){
            if(temp.key.equals(key)){
                if(previous == null){
                    ht[find] = temp.next;
                }else{
                    previous.next = temp.next;
                }
                return;
            }
            previous = temp;
            temp = temp.next;
        }
    }

public static int diamondCount(Stack stack, String str) {
        int dia = 0;

        for(int x = 0; x < str.length(); x++){
            char a = str.charAt(x);

            if(a == '<'){
                stack.push(a);
            }else if(a == '>'){
                if (!stack.isEmpty() && (Character) stack.peek() == '<') {
           stack.pop();
           dia++;
                   }
            }
        }

        return dia;     }
