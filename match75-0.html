public void remove( Integer key ){

             if (idx == -1|| ht [idx]==null){
            return;
        }
          PairNode head = ht[idx];
        if(key.equals(head.key)){


            ht[idx]= head.next;
            return;
        }
           PairNode prev = head;
        PairNode curr = head.next;
       while(curr!= null){
       if(curr.key.equals(key)){
                prev.next=curr.next;
                return;
            }
            prev = curr;

public static int diamondCount(Stack stack, String str) { 
       int diamondCount =0;
           for (int i =0 ; i< str.length(); i++){
           char ch = str.charAt(i);
           if (ch == '<'){
               stack.push(ch);
           }
           else if (ch=='>'){
               if (!stack.isEmpty()){
                   stack.pop();
                   diamondCount++;
               }
           }
       }
    return diamondCount; 
    }

public static String removeConsecDups(String word){
        if (word == null || word.length() == 0){
            return "";
        }


          LinkedListQueue qu = new LinkedListQueue ();
          char x = word.charAt(0);
           qu.enqueue(x);
           for(int i = 1; i<word.length(); i++){
            char y = word.charAt (i);
            if (y != x){
                qu.enqueue(y);
                x=y;
            }
        }
        String result ="";
        while (!qu.isEmpty()){
            result += qu.dequeue();
        }
        return result;
    }
