public void remove( Integer key ){
        int hashindx = hashFunction(key);
        PairNode temp = ht[hashindx];
        if(temp == null){
            return;
        }
        if(temp.key == key){
            ht[hashindx] = temp.next;
            return;
        }
        while(temp.next!= null){
            if(temp.next.key == key){
                temp.next = temp.next.next;
                return;
            }
            temp = temp.next;
        }
    }

private int hashFunction( Integer key ){
        return (key+3)%6;
    }

    public static int diamondCount(Stack stack, String str){
        int diamonds = 0;
        for(int i = 0; i<str.length(); i++){
            char ch = str.charAt(i);
            if(ch == '<'){
                stack.push(ch);
            }


            else if(ch == '>'){
                if(stack.isEmpty() == false && (char)stack.peek() == '<'){
                    stack.pop();
                    diamonds++;
                }
            }
        }
        return diamonds;
    }

public static String removeConsecDups(String word){
        LinkedListQueue queue = new LinkedListQueue();
        for(int i = 0; i<word.length(); i++){
            queue.enqueue(word.charAt(i));
        }
        if(queue.isEmpty()){
            return "";
        }
        char prev = (char) queue.dequeue();
        String result = "" + prev;


        while(queue.isEmpty() == false){
            char current = (char)queue.dequeue();
            if(current!= prev){
                result = result + current;
                prev = current;
            }
        }
        return result;
    }
