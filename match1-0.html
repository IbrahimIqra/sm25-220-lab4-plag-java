private int hashFunction(Integer k) {
    return (k + 3) % 6;
}




public void remove(Integer k) {
    int idx = hashFunction(k);
    if (ht[idx] == null) {
        return; 
    }
    
    PairNode curr = ht[idx];
    PairNode prev = null;
    
    if (curr != null && curr.key.equals(k)) {
        ht[idx] = curr.next;
        return;
    }
    
    while (curr != null && !curr.key.equals(k)) {
        prev = curr;
        curr = curr.next;
    }
    
    if (curr == null) {
        return;
    }
    prev.next = curr.next;
}

public static int diamondCount(Stack s, String str) {
    int cnt = 0;
    
    for(int i = 0; i < str.length(); i++) 
{
        char c = str.charAt(i);
        
        if(c == '.')
 {
            continue;
        }
        if(c == '<')
 {
            s.push(c);
        }
        else if(c == '>')
 {
            if(!s.isEmpty() && (Character)s.peek() == '<') {
                s.pop();
                cnt++;
            }
        }
    }
    return cnt;
}

public static String removeConsecDups(String input) {
    LinkedListQueue queue = new LinkedListQueue();
     


    for(int i = 0; i < input.length(); i++) {
        char currentChar = input.charAt(i);
        
        if(queue.isEmpty()) {
            queue.enqueue(currentChar);
        }
        else if(currentChar != (Character)queue.peek()) {
            queue.enqueue(currentChar);
        }
    }
    
    String result = "";
    while(!queue.isEmpty()) {
        result += queue.dequeue();        
    }
    return result;
}
