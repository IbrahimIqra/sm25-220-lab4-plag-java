 }
 public void remove( Integer key ){
      int num = hashFunction(key);
      PairNode current = ht[num];
      PairNode prev = null;
      while (current != null) {
        if (current.key.equals(key)){
            if (prev == null) {
                ht[num] = current.next;
            }
            else{
                prev.next = current.next;
            }
            return;
        }
        prev = current;
        current = current.next;
     }
  }

public static int diamondCount(Stack stack, String str) {
       int count = 0;
       for (int i = 0; i < str.length(); i++) {
         char ch = str.charAt(i);
         if (ch == '<') {
            stack.push('<');  
         } 
         else if(ch == '>'){
            if(!stack.isEmpty() && stack.peek().equals('<')){
                stack.pop();  
                count++;  
            }
         }
       }
    return count;
    }

    public static String removeConsecDups(String word){
     LinkedListQueue queue = new LinkedListQueue();
     for (int i=0; i<word.length(); i++){
        queue.enqueue(word.charAt(i));
     }
     String output = "";
     if(!queue.isEmpty()){
        char prev = ((Character) queue.dequeue()).charValue();
        output += prev;
        while(!queue.isEmpty()){
            char current = ((Character) queue.dequeue()).charValue();
            if(current != prev){
                output += current;
                prev = current;
            }
        }
     }
    return output;
    }
