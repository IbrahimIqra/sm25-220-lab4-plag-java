 private int hashFunction( Integer key ){
        int index = (key + 3) % ht.length;
        return index;
 }
    
public void remove( Integer key ){


        if (key != null){
            int i = hashFunction(key);
            PairNode current = ht[i] ;
            PairNode prev    = null  ;   
            while ( current != null ) {
                if (current.key.equals(key)) {
                    if (prev == null) {
                        ht[i] = current.next;
                    } 
                    else {
                        prev.next = current.next;
                    }
                    break;
                }


                prev    = current;
                current = current.next;
            }
        }
 }

public static int diamondCount(Stack stack, String str) {
        int count = 0;        
        for (int i = 0; i < str.length(); i++) {
            char ch = str.charAt(i);           
            if (ch == '<') {
                stack.push(ch);
            } 
            else{
                if (ch == '>' && stack.peek()!= null) {


                    stack.pop();
                    count++;
                }
            }
        }        
        return count;       
 }

public static String removeConsecDups(String word){
        LinkedListQueue queue = new LinkedListQueue();
        String result = "";       
        for (int i = 0; i < word.length(); i++) {
            queue.enqueue(word.charAt(i));
        }       
        if (queue.isEmpty()) {
            return result;
        }
        else{


            char prev    = (char)queue.dequeue(); 
            result       += prev; 
            while (!queue.isEmpty()) {
                char current = (char)queue.dequeue() ;               
                if (current != prev) {
                    result += current;
                    prev    = current;
                }
            }
        }
        return result; 
}
