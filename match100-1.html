1)public void remove(Integer key) {


    int index = (key + 3) % 6;                        // Finding right bucket
    PairNode current = hashTable[index];
    PairNode previous = null;
     while (current != null) {
        if (current.key.equals(key)) {     //found the box to remove!
          if (previous == null) {    
             hashTable[index] = current.next;
            } 
          else {
                                                     // Skip this box in the chain
              previous.next = current.next;
            }
            return; 
        }
        previous = current;
        current = current.next;  
    }
}

          stack.push(ch); 
        }
    else if (ch == '>') {
       if (!stack.isEmpty() && (char)stack.peek() == '<') {
          stack.pop();                   //found a complete "<>"
                diamondCount++;
