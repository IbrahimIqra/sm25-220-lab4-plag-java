public static int diamondCount(Stack stack, String str) {
        int count=0;
        for(int i=0;i<str.length();i++){
            char c=str.charAt(i);
            if(c=='<'){
                stack.push(c);
            }
            else if(c=='>'){
                if(!stack.isEmpty()&&stack.peek().equals('<')){
                    stack.pop();
                    count++;
                }
            }
        }
        return count; 
    }

public static String removeConsecDups(String word){
        if(word==null||word.isEmpty()){
            return word;
        }


        LinkedListQueue q=new LinkedListQueue();
        LinkedListQueue r=new LinkedListQueue();//result queue
        for(int i=0;i<word.length();i++){
            char c=word.charAt(i);
            if(q.isEmpty()){
                q.enqueue(c);
                r.enqueue(c);
            }
            else{
                if((char)q.peek()!=c){
                    q.dequeue();
                    q.enqueue(c);
                    r.enqueue(c);
                }
            }
        }
        String result="";
        while(!r.isEmpty()){
            result+=r.dequeue();
        }
        return result;
    }
