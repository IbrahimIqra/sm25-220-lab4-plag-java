private int hashFunction( Integer key ){
        
        return  (key + 3) % ht.length ; 
    }


	
    public void remove( Integer key ){
        int HIndex = hashFunction(key);
        PairNode head = ht[HIndex];

      if(head==null){
       return;
      }

      if(head.key == key){
        ht[HIndex] = ht[HIndex].next;
        return;
      }
      else{
        PairNode curr = head;
        while (curr != null && curr.next != null) {
        if (curr.next.key == key) {


            curr.next = curr.next.next;
            return;
        } 
       
        curr = curr.next; 
    }

      }

    }

public static int diamondCount(Stack stack, String str) {
       int count =0;
       for (int i = 0; i < str.length(); i++) {
        char ch = str.charAt(i);
        if(ch=='<'){
            stack.push(ch);
        }
        else if (ch == '>') {
        if(!stack.isEmpty() && (Character)stack.peek()=='<'){
            stack.pop();
            count++; 
        }

       }
    }
       
    return count; 
    }

public static String removeConsecDups(String word){
	LinkedListQueue queue = new LinkedListQueue();
    for (int i = 0; i < word.length(); i++) {
        char ch = word.charAt(i);
        queue.enqueue(ch);
    }

    if(queue.isEmpty()) {
       return "";

    }
    String finalResult = "";
    char prev = (char) queue.dequeue();
    finalResult+=prev;

    while(!queue.isEmpty()){
        char current = (char) queue.dequeue();

        if(current!=prev){
            finalResult+= current;
            prev = current;
        }
    }

	return finalResult; 
    }
