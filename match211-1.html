public class StackAssgnTaskTester {
    public static int diamondCount(Stack stack, String str) {
        int count=0;
        for (int i=0;i<str.length();i++){
            char ch=str.charAt(i);
            if (ch=='<'){
                stack.push(ch);

            }
            else if (ch=='>'){
                if (!stack.isEmpty()&&(char)stack.peek()=='<'){
                stack.pop();
                count++;
                }
            }
        }

        return count ;
    }

    public static String removeConsecDups(String word){
	LinkedListQueue qu=new LinkedListQueue();
    for (int i=0;i<word.length();i++){
        qu.enqueue(word.charAt(i));

    }
    String result="";
    if(!qu.isEmpty()){
        char prev=(char)qu.dequeue();
        result+=prev;
    
    while(!qu.isEmpty()){
        char current=(char)qu.dequeue();

        if(current!=prev){
            result+=current;
            prev=current;
        }

    }
    }
    return result;
