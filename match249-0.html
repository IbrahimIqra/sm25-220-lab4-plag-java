 public static int diamondCount(Stack stack, String str) {
        int dcount=0;
        
        for(int i=0; i<str.length();i++){
           char curr=str.charAt(i);
           
           if(curr=='<'){
             stack.push(curr);}
             else if(curr=='>'){
               if(!stack.isEmpty()){
                 char temp=(Character)stack.peek();
                 if(temp=='<'){
                   stack.pop();
                   dcount++;}}
             }
           }
        return dcount;
    }

public static String removeConsecDups(String word){
LinkedListQueue task2= new LinkedListQueue();
char check= word.charAt(0);
task2.enqueue(check);
for(int i=0; i<word.length();i++){
  char temp=word.charAt(i);
  if(check!=temp){
    task2.enqueue(temp);
  check=temp;}
    }
String output="";
while(!task2.isEmpty()){
  output+=task2.dequeue();}
return output;}
