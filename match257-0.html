    public void remove( Integer key ){
        int bucket = hashFunction(key);
        
        PairNode current = ht[bucket];
        PairNode previous = null;
            
            while (current != null) {
            if (current.key.equals(key))
            {
                if (previous == null)
                {
                    ht[bucket] = current.next;
                }
                else
                {
                    previous.next = current.next;
                }
                return;
            }
            previous = current;
            current = current.next;

    public static int diamondCount(Stack stack, String str) {
        int diamonds = 0;
        for (int i = 0; i < str.length(); i++) // we gotta ignore the dots
        {


            char ch = str.charAt(i);
            if (ch == '<')
            {
                stack.push('<');
            }
            else if (ch == '>')
            {
                if (!stack.isEmpty()) // If stack is empty, simply ignore this '>'
                {
                    stack.pop();
                    diamonds++;
                }
            }
        }
        return diamonds;
    }

    public static String removeConsecDups(String word){
        


        LinkedListQueue queue = new LinkedListQueue();
        
        for(int i = 0; i < word.length(); i++)
        {
            queue.enqueue(word.charAt(i));
        }
        
        String result = "";


        Character last = null;

        while (!queue.isEmpty())
        {
            char current = (char) queue.dequeue();
            
            if (last == null || current != last)
            {
                result = result + current;
                last = current;
            }
        }
        return result;
    }
