private int hashFunction( Integer key ){
        return (key+3) % 6;
}
public void remove( Integer key ){
        int x=hashFunction(key);
        PairNode pred=null;
        PairNode remNode=ht[x]; // eita ami remove korbo
        


        while(remNode!=null){
            if(remNode.key.equals(key)){
                if(pred==null){ // matha khali jodi hoi means LL er first e asi
                    ht[x]=remNode.next; // head ta akhon remNode er porer node ke dhortese 
                }
                else{
                    pred.next=remNode.next; // eikahane ami remNode taake skip korlam
                }
                return;
            }
            pred=remNode; // remNode e pred pointer ke update korlam
            remNode=remNode.next; // rem keo update korlam
        }
}

public static int diamondCount(Stack stack, String str){    
    int D_count = 0;
    for (int i= 0; i<str.length(); i++) {
        char ch=str.charAt(i);
        if (ch=='<'){
            stack.push(ch);
        } 
        else if(ch=='>'){
            if(!stack.isEmpty() && stack.peek().equals('<')){ // khali and opore peek kore dekhbo je matching bracket ase or nai
                stack.pop(); // jodi pai match tahole pop korbo
                D_count++; // akta diamond paisi so increment
            }
        }
    }
    return D_count;
}

public static String removeConsecDups(String word){ 
    LinkedListQueue queue =new LinkedListQueue(); 
    for(int i=0; i<word.length(); i++){
            queue.enqueue(word.charAt(i));
        }
    String result ="";
    if(!queue.isEmpty()){ // jodi na khali thake
        char previous=(char)queue.dequeue();
        result+=previous;
        while(!queue.isEmpty()){
            char current=(char)queue.dequeue();
            if(current!=previous){
                result+=current;
                previous=current;
            }
        }
    }
    return result;

    }
