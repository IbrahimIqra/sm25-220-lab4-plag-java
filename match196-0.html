  public void remove( Integer key ){
     int indx = hashFunction(key) ;
     PairNode head = ht[indx] ;
     //check if exists then change or do nothing
      if(head == null) return;
      if(head.key.equals(key)){
          ht[indx] = head.next ;
          return;
      }
      while(head.next != null){
          if(head.next.key.equals(key)){
              head.next = head.next.next ;
              return;
          }
          head = head.next ;
      }
  }

public static int diamondCount(Stack stack, String str) {
   int count = 0 ;
   for(int i=0 ; i< str.length() ; i++){
       char sym = str.charAt(i);
       if(sym == '<'){
           stack.push(sym);
       }
       if(sym == '>'){
           if(!stack.isEmpty() && stack.peek().equals('<')){
               count++;
               stack.pop() ;
           }
       }
   }
   return count;
}

   public static String removeConsecDups(String word) {


     //try with single loop
       LinkedListQueue l1 = new LinkedListQueue();
       for (int i = 0; i < word.length(); i++) {
           l1.enqueue(word.charAt(i));
       }
       String s= "" ;
       Character prev = null ;
       while (!l1.isEmpty()){
           Character curr = (Character) l1.dequeue() ;
           if(curr!= null && prev!=curr){
               s += curr ;
               prev = curr;
           }
       }
       return s ;
   }
