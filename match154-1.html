 public static String removeConsecDups(String word){
        LinkedListQueue q = new LinkedListQueue();


    for (int i = 0; i < word.length(); i++) {
        q.enqueue(word.charAt(i));
    }


    String finalStr = "";
    if (!q.isEmpty()) {
        char last = (char) q.dequeue();
        finalStr = finalStr + last;




        while (!q.isEmpty()) {
            char now = (char) q.dequeue();
            if (now != last) {
                finalStr = finalStr + now;
            }
            last = now;
        }
    }


    return finalStr;
}

public static int diamondCount(Stack stack, String str) {
        int diamonds = 0;


        for (int i = 0; i < str.length(); i++) {
            char ch = str.charAt(i);


            if (ch == '<') {
                stack.push('>');
            } else if (ch == '>') {
                if (!stack.isEmpty()) {
                    Object top = stack.peek();
                    if (top.equals('>')) {
                        stack.pop();
                        diamonds++;
                    }
                }
            }


        }


        return diamonds;
    }
