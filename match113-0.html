// Assignment-01

private int hashFunction( Integer key ){


        return (key + 3) % ht.length;
}
public void remove( Integer key ){
        int idx = hashFunction(key);
        PairNode temp = ht[idx];

        if(temp == null){
            return;
        }

        if(temp.key.equals(key)) {
            ht[idx] = temp.next;
            return;
        }

        PairNode prev = temp;


        temp = temp.next;

        while(temp != null) {
            if(temp.key.equals(key)) {
                prev.next = temp.next; 
                return;
            }
            prev = temp;
            temp = temp.next;
        }
}

public static int diamondCount(Stack stack, String str) {
        int count = 0;
        for(int i = 0; i<str.length(); i++){
            char ch = str.charAt(i);
            if(ch == '<'){
                stack.push(ch);
            }
            else if(ch == '>'){
                if(!stack.isEmpty()){
                    stack.pop();
                    count++;
                }
            }
        }
        return count;
}

public static String removeConsecDups(String word){


        LinkedListQueue q = new LinkedListQueue();
        for(int i = 0; i<word.length(); i++){
            q.enqueue(word.charAt(i));
        }
        String str = "";
        char temp = '\0';
        while(!q.isEmpty()){
            char current = (char)q.dequeue();
            if(current != temp){
                str += current;
                temp = current;
            }
        }
        return str;
}
