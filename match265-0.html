public static int diamondCount(Stack stack, String str) {
        // To Do

        int dcount = 0;

        for(int i = 0; i < str.length(); i++){

            char ch = str.charAt(i);

            if(ch == '<'){

                stack.push(ch);
            }else if(ch == '>'){

                if(!stack.isEmpty() && (char)stack.peek() == '<'){

                    stack.pop();
                    dcount ++;
                }
            }
        }

        return dcount; // Delete this line
    }





public static String removeConsecDups(String word){
    LinkedListQueue llq = new LinkedListQueue();

    // Enqueue all characters
    for (int i = 0; i < word.length(); i++) {
        llq.enqueue(word.charAt(i));
    }

    // Prepare result string
    String output = "";

    if (!llq.isEmpty()) {
        char dx = (char) llq.dequeue();
        output += dx;

        while (!llq.isEmpty()) {
            char curr = (char) llq.dequeue();

            if (curr != dx) {
                output += curr;
                dx = curr;
            }
        }
    }

    return output;
}
