public static String removeConsecDups(String word) {
        LinkedListQueue charQueue = new LinkedListQueue();
        for (int i = 0; i < word.length(); i++) {
            charQueue.enqueue(word.charAt(i));
        }

        String final_str = "";
        char lastChar = '\0';
        while (!charQueue.isEmpty()) {
            char currentChar = (char) charQueue.dequeue();
            if (currentChar != lastChar) {
                final_str += currentChar;
                lastChar = currentChar;
            }
        }

        return final_str;
    }
