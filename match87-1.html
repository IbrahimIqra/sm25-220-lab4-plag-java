public void remove(Integer key){
    int index = hashFunction(key);

    if (ht[index] == null){
        return;
    }

    if (ht[index].key.equals(key)) {
        ht[index] = ht[index].next;
        return;
    }

    PairNode prev = ht[index];


    PairNode curr = ht[index].next;

    while (curr != null) {
        if (curr.key.equals(key)) {
            prev.next = curr.next; 
            return;
        }
        prev = curr;
        curr = curr.next;
    }
}

   public static int diamondCount(Stack stack, String str) {
        int count = 0;
    
        for (int i = 0; i < str.length(); i++) {
            char ch = str.charAt(i);
    
            if (ch == '<') {
                stack.push(ch);
            } else if (ch == '>') {
                // Try to match a '<' from the top
                if (!stack.isEmpty() && (char)stack.peek() == '<') {
                    stack.pop();
                    count++;
                }
            }
        
        }
    
        return count;
    }

    public static String removeConsecDups(String word) {
        if (word == null || word.length() <= 1) {
            return word;
        }



        LinkedListQueue q = new LinkedListQueue();
        for (int i = 0; i < word.length(); i++) {
            q.enqueue(word.charAt(i));
        }

        String result = "";


        char prev = (char) q.dequeue();
        result += prev;

        while (!q.isEmpty()) {
            char current = (char) q.dequeue();
            if (current != prev) {
                result += current;
                prev = current;
            }
        }

        return result;
    }
