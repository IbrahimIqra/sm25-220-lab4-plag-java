private int hashFunction(Integer key) {
    return (key + 3) % 6;
}

public void remove(Integer key) {
    int idx = hashFunction(key);
    PairNode current = ht[idx];
    PairNode previous = null;

    while (current != null) {
        if (current.key.equals(key)) {
            if (previous == null) {
                ht[idx] = current.next;
            } 
else {
                previous.next = current.next;
            }
            return; // Exit after removing
        }
        previous = current;
        current = current.next;
    }
}

public static int diamondCount(Stack stack, String str) {
    int count = 0;

    for (int i = str.length() - 1; i >= 0; i--) {


        char ch = str.charAt(i);

        if (ch == '>') {
            stack.push('>');
        }
 else if (ch == '<') {
            if (!stack.isEmpty() && (char)stack.peek() == '>') {
                stack.pop(); 
                count++;                }
        }
      }

    return count;
}

public static String removeConsecDups(String word){
    LinkedListQueue inputQueue = new LinkedListQueue();


    LinkedListQueue filteredQueue = new LinkedListQueue();
     for (int i = 0; i < word.length(); i++) {
        inputQueue.enqueue(word.charAt(i));
    }
    Character prev = null;



    while (!inputQueue.isEmpty()) {
        char current = (char) inputQueue.dequeue();
        if (prev == null || current != prev) {
            filteredQueue.enqueue(current);
            prev = current;
        }
    }
    String result = "";
    while (!filteredQueue.isEmpty()) {
        result += filteredQueue.dequeue(); 
 }
    return result;
}
