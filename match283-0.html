 public void remove( Integer key ){
      int index = hashFunction(key);
      PairNode current = ht[index];
      if(current == null){
      return;
      }
      if(current.key.equals(key)){
       current = current.next;


        ht[index] = current;
        return;
      }
      PairNode tail = current;
      while(tail.next!= null){
        if(tail.next.key.equals(key)){
        tail.next  = tail.next.next ;
        return;
        }
        else{
       tail = tail.next;
        }
     }
      return;
      }

 public static int diamondCount(Stack stack, String str) {
       
      int count = 0;
      for(int i=0; i<str.length() ; i++){
      char ch = str.charAt(i);
      if(ch == '<'){
      stack.push(ch);
      }
      else if(ch=='>'){


        if(stack.isEmpty()== false && (char)stack.peek() == '<' ){
        stack.pop();
         count++;   
        }
      }
       
      }
      

        return count;
    }

    public static String removeConsecDups(String word){

     LinkedListQueue q = new LinkedListQueue();
      for(int i=0; i<word.length(); i++){

      q.enqueue(ch);
      }
      String s = "";
      
      if(q.isEmpty()==false){
     char ch =  (char)q.dequeue();
     s+=ch;
     while(q.isEmpty()==false){
     char aws = (char)q.dequeue();
     if(aws != ch){
     s+=aws;
     }
     ch = aws;
     }
     return s;
      }
