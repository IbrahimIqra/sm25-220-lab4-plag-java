public static int diamondCount(Stack stack, String str) {
       // To Do
       int count = 0;
       for(int i = 0; i<str.length(); i++){


           char bit = str.charAt(i);
          
           if(bit == '<'){


               stack.push(bit);


           }
           else if(bit == '>'){
               if (!stack.isEmpty() && (char)stack.peek()=='<'){
                   stack.pop();
                   count++;
               }
           }
       }




       return count;

   if(!que.isEmpty()){

       while(!que.isEmpty()){
