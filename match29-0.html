  private int hashFunction( Integer key ){


        return  (key+3)% ht.length;
    }
    
    public void remove( Integer key ){
        int index = hashFunction(key);
        PairNode current = ht[index];
        PairNode prev = null ;
        while ( current != null ){
            if (current.key.equals( key )){
                if(prev == null ){
                    ht[index] = current.next ;
                }else{
                    prev.next = current.next ;
                }
                return;
            }
            prev = current ;
            current = current.next ;
        }
    }

public static int diamondCount(Stack stack, String str) {
        int diamonds = 0 ;

        for (int i = 0 ; i < str.length() ; i++){
            char ch = str.charAt(i);

            if(ch == '<'){
                stack.push(ch);
            }else if(ch == '>'){
                if(!stack.isEmpty() && (Character) stack.peek()=='<'){
                    stack.pop();
                    diamonds++;
                }
            }
        }
        return diamonds;
    }

 public static String removeConsecDups(String word){


        LinkedListQueue queue = new LinkedListQueue();
        for(int i = 0 ; i < word.length() ; i ++){
            queue.enqueue(word.charAt(i));
        }
        String out = "";
        if(!queue.isEmpty()){
            char prev = (char) queue.dequeue();
            out+= prev ;
            while(!queue.isEmpty()){
                char current = (char) queue.dequeue();
                if(current != prev ){
                    out+= current ;
                    prev = current ;
                }
            }
        }
         return  out ;
    }
