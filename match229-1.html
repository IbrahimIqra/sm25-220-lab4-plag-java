public static int diamondCount(Stack stack, String str) {
        int d=0;
        
        for(int i=0;i<str.length();i++){
           char c=str.charAt(i);
           if(c=='<'){


            stack.push('<');
           }
            
            else if(c=='>'){
                if(!stack.isEmpty() &&(Character)stack.peek()=='<'){
                    stack.pop();
                    d++;
                }
            
            }
        }
        return d;
        
    }

public static String removeConsecDups(String word){
	
    LinkedListQueue nl=new LinkedListQueue();
    int i=0;
    while(i<word.length()){
        nl.enqueue(word.charAt(i));
        i++;

    }
    String fin="";
    Character prev=null;
    while(!nl.isEmpty()){
        char now=(Character)nl.dequeue();
        if(!Character.valueOf(now).equals(prev)){
            fin=fin+now;
            prev=now;
        }
    }
	return fin; 
    }
