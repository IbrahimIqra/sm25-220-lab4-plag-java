private int hashFunction( Integer key ){
        return (key+3)%ht.length;
    }


public void remove( Integer key ){
        int index = hashFunction(key);


        PairNode prev = null;
        PairNode curr = ht[index];


        while (curr!=null) {
            if (curr.key == key) {


                if (prev==null) {
                    ht[index] = curr.next;
                } else {
                    prev.next = curr.next;
                }
                return;
            }
            prev = curr;
            curr = curr.next;
        }
    }

public static int diamondCount(Stack stack, String str) {
        int diamondCount = 0;
        for (int i = 0; i < str.length(); i++) {
            char c = str.charAt(i);
            if (c == '>') {


                stack.push(c);
            }
            else if (c == '>') {
                if (!stack.isEmpty()&&(char)stack.peek == '<') {
                    stack.pop();
                    diamondCount++;
                }
            }
        }
        return diamondCount;
    }

public static String removeConsecDups(String word){
            LinkedListQueue q = new LinkedListQueue();


        for (int i = 0; i < word.length(); i++) {
            q.enqueue(word.charAt(i));
        }


        String ans = "";


        while (!q.isEmpty()) {
            char curr = (char)q.dequeue();
            ans+=curr;


            while (!q.isEmpty()&&(char)q.peek()==curr) {
                q.dequeue();
            }
        }
            return ans; 
    }
