    public void remove( Integer key ){

    PairNode newNode = null;

    for (;n != null;) {

    public static String removeConsecDups(String word){
	// TODO

    LinkedListQueue Object = new LinkedListQueue();
    for (int a = 0; a < word.length(); a++) {
        Object.enqueue(word.charAt(a));
    }

    String adding = "";

    if (!Object.isEmpty()) {
        char at = (char) Object.dequeue();
        adding += at;

        for (;!Object.isEmpty();) {
            char current = (char) Object.dequeue();
            if (current != at) {
                adding += current;
                at = current;
            }
        }
    }
