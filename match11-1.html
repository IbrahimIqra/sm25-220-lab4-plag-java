 public static String removeConsecDups(String word) {
    LinkedListQueue lol = new LinkedListQueue();

    // Enqueue all characters
    for (int i = 0; i < word.length(); i++) lol.enqueue(word.charAt(i));

    if (lol.isEmpty()) return "";

    String newS = "";


    char prev = (char) lol.dequeue(); 
    newS += prev;

    while (!lol.isEmpty()) {
        char current = (char) lol.dequeue();
        if (current != prev) {  
            newS += current;
            prev = current;
        }
    }

    return newS;
}
