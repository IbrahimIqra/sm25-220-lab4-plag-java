public void remove(Integer key) {
    int location = hashFunction(key);
    PairNode current = ht[location];

    if (current != null && current.key.equals(key)) {
        ht[location] = current.next;
        return;
    }

    while (current.next!= null) {
        if (current.next.key.equals(key)) {
            current.next = current.next.next;
            return;
        }


        current = current.next;
    }
}

public static int diamondCount(Stack stack, String str) {
        
    int count = 0;
    for (int k = 0; k < str.length(); k++) {
        char ch = str.charAt(k);
        if (ch == '<') {
            stack.push(ch);
        } else if (ch == '>') {
            if (!stack.isEmpty() && (char) stack.peek() == '<') {


                stack.pop();
                count++;
            }
        }
    
    }
    return count;


    }

public static String removeConsecDups(String word) {
    
    LinkedListQueue q = new LinkedListQueue();

    
    for (int j = 0; j < word.length(); j++) {
        q.enqueue(word.charAt(j));
    }

    
    String f = "";
    if (!q.isEmpty()) {
        char pre= (char) q.dequeue();
        f += pre;

        while (!q.isEmpty()) {
            char curr = (char) q.dequeue();
            if (curr != pre) {
                f += curr;
                pre = curr;
            }
        }
    }

    return f;
}
