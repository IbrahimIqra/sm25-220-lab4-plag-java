  private int hashFunction( Integer key ){
        return (key+3)%ht.length;
    }

    public void remove( Integer key ){
        int idx=hashFunction(key);
    PairNode currNode=ht[idx];
    PairNode prevNode=null;

    while(currNode!=null){
        if(currNode.key==key){
            if (prevNode==null){
                ht[idx]=currNode.next;
                return;
            }else{
                prevNode.next=currNode.next;
                return;
            }
        }


        prevNode=currNode;
        currNode=currNode.next;
    }
}

 public static int diamondCount(Stack stack, String str) {
         int count=0;
        for(int i=0; i<str.length(); i++){
        char c=str.charAt(i);

        if(c=='<'){


            stack.push(c);
        }else if(c=='>'){
            if(!stack.isEmpty() && (char)stack.peek()=='<') {
                stack.pop();
                count++;
            }
        }
    }
   return count;
 }

 public static String removeConsecDups(String word){
    LinkedListQueue queue=new LinkedListQueue();
    String s1="";


    for(int i=0;i<word.length();i++){
        queue.enqueue(word.charAt(i));
    }
    char firstCh=(char)queue.dequeue();
    s1+=firstCh;


    while(!queue.isEmpty()){
        char temp=(char)queue.dequeue();
        if(temp!=firstCh) {
            s1+=temp;
            firstCh=temp;
        }else{
            continue;
       }
    }
   return s1;
 }
