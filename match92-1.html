            }

               temp= (char)stack.pop(); 
               if(temp =='<'){
                    count++;

    public static String removeConsecDups(String word){
    LinkedListQueue lq =new LinkedListQueue();
    String s ="";
    char rear =0;


    for(int i=0;i<word.length();i++){
        char ch =word.charAt(i);
        if(lq.isEmpty()){
            lq.enqueue(ch);
            rear =ch;
        }
        else{
            char temp =rear;
            if(ch==temp){
                continue;
            }
            else{
                lq.enqueue(ch);
                rear =ch;
            }
        }
    }
    while(!lq.isEmpty()){
    s+=lq.dequeue();
        }

    return s;
    }
