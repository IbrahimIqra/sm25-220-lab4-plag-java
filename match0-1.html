//Task 01

private int hashFunction( Integer key ){
    int h = (key + 3) % ht.length;


    return h;
}

public void remove( Integer key ){
    int h = hashFunction(key);
    if(ht[h] != null){
        PairNode current;
        PairNode prev = null;
        for(current = ht[h]; current != null; current = current.next){
            if(current.key == key){
                if(prev == null){
                    ht[h] = current.next;
                }
                else {
                    prev.next = current.next;
                }
                break;
            }
            prev = current;
        }
    }
}

public static int diamondCount(Stack stack, String str) {


    int count = 0;
    // iterate the string
    for(int a=0; a< str.length(); a++){
        if(str.charAt(a) == '.'){
            continue;
        }
        else if(str.charAt(a) == '<'){
            stack.push("<");
        }
        else if(str.charAt(a) == '>' && stack.peek() != null){
            String last_elem = stack.peek().toString();
            if(last_elem.equals("<")){
                count++;
                stack.pop();
            }
        }
    }

    return count; 
}

public static String removeConsecDups(String word){
    LinkedListQueue queue = new LinkedListQueue();
    String new_string = "";
    for(int a=0; a< word.length(); a++){
        if(queue.isEmpty()){
            queue.enqueue(word.charAt(a));
        }
        else if((char)queue.peek() == word.charAt(a)){
            continue;
        }
        else {
            new_string += queue.dequeue().toString();
            queue.enqueue(word.charAt(a));
        }
    }
    new_string += queue.dequeue().toString();
    return new_string; 
}
