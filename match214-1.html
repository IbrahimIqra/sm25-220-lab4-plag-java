 public void remove( Integer key ){
        int index = hashFunction(key);
        PairNode c = ht[index];
        PairNode p = null;
        while(c!=null){
            if(c.key.equals(key)){
               if(p==null){
                ht[index] = c.next;
               }
               else{
                p.next = c.next;
               } 
              return;
            }
          p = c;
          c = c.next;
        }

  public static int diamondCount(Stack stack, String str) {
      int count = 0;

      for(int i = 0; i < str.length(); ++i) {
         char c = str.charAt(i);
         if (c == '<') {
            stack.push(c);
         } else if (c == '>' && !stack.isEmpty()) {
            stack.pop();
            count++;
         }
      }

      return count;
   }

public static String removeConsecDups(String word){
        LinkedListQueue queue = new LinkedListQueue();
        for(int i = 0; i < word.length(); i++){
            queue.enqueue(word.charAt(i));
        }
        String newString = "";
        if(!queue.isEmpty()){
            char head = (char)queue.dequeue();
            newString+=head;
            while(!queue.isEmpty()){
                char tail = (char) queue.dequeue();
                if(tail!=head){
                    newString+=tail;
                    head = tail;
                }
            }


        }
        return newString;
}
