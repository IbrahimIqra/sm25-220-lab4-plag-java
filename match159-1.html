public static int diamondCount(Stack stack, String str) {
    int diamond = 0;

    for(int i=0; i<str.length(); i++) {
        char c=str.charAt(i);

          if (c=='>') {
             Object dot1=null;
             Object dot2 =null;
              Object left =null;

             if (!stack.isEmpty()) dot1 =stack.pop();
             if (!stack.isEmpty()) dot2 =stack.pop();
              if (!stack.isEmpty()) left =stack.pop();

              if (dot1 !=null && dot2!=null && left!=null && dot1.equals('.') && dot2.equals('.') && left.equals('<')) {
                
                 diamond++;
              } else {

                 if (dot1!=null) stack.push(dot1);
                 if (dot2!=null) stack.push(dot2);
                 if (left!=null) stack.push(left);
                 
              }
           } else {
           
            stack.push(c);
         }
     }

     return diamond;
 }

public static String removeConsecDups(String word) {
    LinkedListQueue queue =new LinkedListQueue();
        for(int i = 0; i < word.length(); i++) {
           queue.enqueue(word.charAt(i));
     }

    String result ="";
    if (!queue.isEmpty()) {
        char prev=(char) queue.dequeue();
        result=result+prev;

        while (!queue.isEmpty()) {
            char cur=(char) queue.dequeue();
            if (cur!=prev) {
                result=result+cur;
                 prev =cur;
            }
        }
     }

    return result;
}
