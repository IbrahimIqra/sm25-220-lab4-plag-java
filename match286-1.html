public static int diamondCount(Stack stack, String str) {
        int count=0;
        for(int i=0;i<str.length();i++)
        {
            char ch=str.charAt(i);
            if(ch=='<')
            {
                stack.push(ch);
            }
            else if(ch=='>')
            {
                if(!stack.isEmpty())
                {
                    stack.pop();
                    count++;
                }   
            }
        }
        return count;
    }

public static String removeConsecDups(String word){

        LinkedListQueue q1=new LinkedListQueue();
        for(int i=0;i<word.length();i++)
        {
            q1.enqueue(word.charAt(i));
        }
        String str="";
        if(!q1.isEmpty())
        {
            char prev=(char)q1.dequeue();
            str+=prev;


            while(!q1.isEmpty())
            {
                char temp=(char)q1.dequeue();
                if(temp!=prev)
                {
                    str+=temp;
                    prev=temp;
                }
            }
        }
