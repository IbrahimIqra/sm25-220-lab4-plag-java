    private int hashFunction( Integer key ){


        return (key + 3) % ht.length;   
    }



    public void remove( Integer key ){
        int index = hashFunction(key);
        PairNode current = ht[index];
        
        if (current == null) {
          return;
        }
        
        if (current.key.equals(key)) {
          ht[index] = current.next;
          return;
        
        }
        
     PairNode previous = current;


     current = current.next;
     
     while (current != null) {
       if (current.key.equals(key)) {          
          previous.next = current.next;
           return;
       }
       previous = current;
       current = current.next;
          
          
     }   
          
          
  }

 public static int diamondCount(Stack stack, String str) {
        int heeracount = 0;

        for (int i = 0; i < str.length(); i++){
          char current = str.charAt(i);
        
        
        if (current == '<' ) {
           stack.push(current);
        }
        else if (current == '>') {
          
          if (!stack.isEmpty()) {


            stack.pop();
            heeracount++;
          }
        }
    }
        return heeracount;
 }

 public static String removeConsecDups(String word) {
        LinkedListQueue queue = new LinkedListQueue();
        String result = "";

        for (int i = 0; i < word.length(); i++) {
           char currentchar = word.charAt(i);

           if (queue.isEmpty() || ((Character)queue.peek()) != currentchar) {
               queue.enqueue(currentchar); 
          }
       }
        while (!queue.isEmpty()) {
            result += queue.dequeue();  
       }
         return result;
  }
