 public static int diamondCount(Stack stack, String str) {
      int count =0;
      
      for(int i=0;i<str.length();i++)
      {
        char ch=str.charAt(i);
        
        if(ch =='.')
        {
          continue;
        }
        if(ch == '<')
        {
          stack.push(ch);
        }
        else if(ch == '>')
        {
          if(!stack.isEmpty() && (Character)stack.peek()=='<')
          {
            stack.pop();
            count++;
          }
        }
      }
      return count;
    }

 public static String removeConsecDups(String word){
      LinkedListQueue a = new LinkedListQueue();
    char lastChar = '\0';
     


    for(int i = 0; i < word.length(); i++)
    {
        char ch = word.charAt(i);
        
        if(a.isEmpty()) 
        {
            a.enqueue(ch);
            lastChar = ch; 
        }
        else if(ch != lastChar) 
        {
            a.enqueue(ch);
            lastChar = ch; 
        }
    }
    
    String b = "";
    while(!a.isEmpty()) {
        b += a.dequeue();        
    }
    return b;
}
