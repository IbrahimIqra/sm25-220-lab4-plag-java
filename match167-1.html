public static int diamondCount(Stack stack, String str) {
    int t=0;
  for(int i=0;i<str.length();i++){
          char ch=str.charAt(i);
                 if(ch==('<')){
              stack.push(ch);

                }  else if(ch=='>'){

              if(stack.peek()!=null && (!stack.isEmpty())){
                  
 char x = (char) stack.peek();
                  if (x== '<') {


                      stack.pop();
                         t++;
                  }
                        }
                                }
      }

    return t;
}

public static String removeConsecDups(String word){
    LinkedListQueue q=new LinkedListQueue();
       for(int i=0;i<word.length();i++){

        q.enqueue(word.charAt(i));

           }





     String s="";
           char prev=(char)q.dequeue();
    s+=prev;

    while(!q.isEmpty()){
        char cur=(char)q.dequeue();
        if(cur!=prev){
            s+=cur;
            prev=cur;
                 }
    }
    return s;
}
