private int hashFunction(Integer m){
  return (m+3)%ht.length;
}


public void remove(Integer m){
  int px=hashFunction(m);
  PairNode r1=ht[px];
  PairNode pv=null;
  while(r1!=null){
    if(r1.key==m){
      if(pv==null){
        ht[px]=r1.next;
      }
      else{
        pv.next=r1.next;
      }
      return;
    }
    pv=r1;
    r1=r1.next;
    
  }
}

public static int diamondCount(Stack s,String r){
  int trk=0;
  for (int i=0;i<r.length();i++){
    char m=r.charAt(i);
    if(m=='<'){
      s.push(m);
    }
    else if(m=='>'){
      if(!s.isEmpty()&&(char)s.peek()=='<'){
        s.pop();
        tk++;
      }
    }
    else if(m=='.'){
      continue;
    }
    else{
      continue;
    }
  }
  return trk; 
}
