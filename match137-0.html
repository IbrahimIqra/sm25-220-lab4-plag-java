        // To Do
        int total=0;
        for(int i=0; i<str.length(); i++){
            char c=str.charAt(i);
            if(c=='<'){
                stack.push(c);
            }else if(c=='>'){
                if(!stack.isEmpty() && stack.peek().equals('<')){

                    total++;

public class RemoveConsecutiveDuplicates{

    public static String removeConsecDups(String word){
	// TODO
	// You MUST create a LinkedListQueue Object to solve this task
    LinkedListQueue a= new LinkedListQueue();
    for(int i=0; i<word.length(); i++){
        a.enqueue(word.charAt(i));
    }
    String finalresult="";
    Character prev=null;
    while(!a.isEmpty()){
        Character curr= (Character)a.dequeue();
        if(prev==null || !prev.equals(curr)){
            finalresult= finalresult + curr;
            prev=curr;
        }
        return finalresult;
    }

	return null; //remove this line once you're ready
    }
