//Stack: Diamond Count:
public class StackAssgnTaskTester {


    public static int diamondCount(Stack stack, String str) {
    int result = 0; 
for (int i = 0; i < str.length(); i++) {
        char var = str.charAt(i); 

        if (var == '<') {
            
            stack.push(var); 
        } 
        else if (var == '>') {
            if (!stack.isEmpty()) {
                char top = (Character) stack.peek(); 
                if (top == '<') {
                    stack.pop(); 
                    result++;
                }
            }
        }
        
    }

    return result;
}


//Queue: Remove Consecutive Duplicates
public class RemoveConsecutiveDuplicates{


 public static String removeConsecDups(String word){
    LinkedListQueue list = new LinkedListQueue();
    for (int i = 0; i < word.length(); i++) {
        list.enqueue(word.charAt(i)); 
    }

    String output = "";
    if (!list.isEmpty()) {
        char var1 = (Character) list.dequeue(); 
        output = output + var1; 
        while (!list.isEmpty()) {
            char var2 = (Character) list.dequeue(); 

            if (var2 != var1) {
                output = output + var2; 
                var1 = var2; 
            }
        }
    }

    return output;
}
