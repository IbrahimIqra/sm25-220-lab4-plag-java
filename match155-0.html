private int hashFunction(Integer key) {


    return (key + 3) % ht.length;
}


public void remove(Integer key) {
    int idx = hashFunction(key);
    PairNode current = ht[idx];
    PairNode previous = null;


    while (current != null) {
        if (current.key.equals(key)) {
            if (previous == null) {
                ht[idx] = current.next;
            } else {
                previous.next = current.next;
            }
            return;
        }
        previous = current;
        current = current.next;
    }
}

public static int diamondCount(Stack stack, String str) {
    int count = 0;


    for (int i = 0; i < str.length(); i++) {
        char ch = str.charAt(i);


        if (ch == '<') {


            stack.push('<');
        } else if (ch == '>') {
            if (!stack.isEmpty() && (char) stack.peek() == '<') {
                stack.pop();
                count++;
            }
        }
    }


    return count;
}

public static String removeConsecDups(String word) {
    LinkedListQueue queue = new LinkedListQueue();


    for (int i = 0; i < word.length(); i++) {
        queue.enqueue(word.charAt(i));
    }


    if (queue.isEmpty()) {
        return "";
    }


    char prev = (char) queue.dequeue();
    String result = "" + prev;


    while (!queue.isEmpty()) {
        char current = (char) queue.dequeue();


        if (current != prev) {
            result = result + current;
            prev = current;
        }
    }


    return result;
}
