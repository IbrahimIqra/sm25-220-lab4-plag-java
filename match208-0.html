        int diamond = 0;

        for(int i = 0 ; i < str.length() ; i++)
        {
            char ch = str.charAt(i);

                stack.push('<');

                if (!stack.isEmpty() && (char)stack.peek() == '<')

 public static String removeConsecDups(String word){
        //TODO
        // You MUST create a LinkedListQueue Object to solve this task
        LinkedListQueue que = new LinkedListQueue();

        if(word == null || word.length() == 0)
            return word;

        char prev = word.charAt(0);
        que.enqueue(prev);

        String result = ""+prev;



        for (int i = 0 ; i < word.length() ; i++)
        {
            char curr = word.charAt(i);
            if(curr != prev)
            {
                que.enqueue(curr);
                result += curr;
                prev = curr;
            }
        }
	return result; //remove this line once you're ready
    }
