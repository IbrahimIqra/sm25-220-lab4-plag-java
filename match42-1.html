 private int hashFunction( Integer key ){

      return (key + 3) % ht.length;
    }


    public void remove( Integer key ){
    int ind = hashFunction(key);
    PairNode current = ht[ind];
    PairNode prev = null;

    while(current != null){
        if(current.key.equals(key)){
            if(prev == null){
                ht[ind] = current.next; 
            } 
            else {


                prev.next = current.next; 
            }
            return;
        }
        prev = current;
        current = current.next;
  
    }

}

public static int diamondCount(Stack stack, String str) {
        int count = 0;

        for (int i = 0; i < str.length(); i++) {
            char ch = str.charAt(i);

            if (ch == '<') {
                stack.push(ch); // Push as Character, not Integer
            } 
            else if (ch == '>') {
                if (!stack.isEmpty()) {
                    Object topObj = stack.peek();

                    if ((Character) topObj == '<') {


                        stack.pop();
                        count++;
                    }
                }
            }
        }

        return count;
    }

    public static String removeConsecDups(String word){

    LinkedListQueue newQ = new LinkedListQueue();
    

    for (int i = 0; i < word.length(); i++) {
        newQ.enqueue(word.charAt(i));
    }

    String result = " ";
    char prev = ' '; 
    
    while (!newQ.isEmpty()) {
        char current = (char) newQ.dequeue();
        if (current != prev) {
            result += current;
            prev = current;
        }
    }
    
    return result;
}
