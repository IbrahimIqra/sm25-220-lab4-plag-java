private int hashFunction( Integer key ){
        int temp = (key+3)%ht.length;
        return temp;
    }

    public void remove( Integer key ){
        int index = hashFunction(key);
        PairNode current = ht[index];
        PairNode prev = null;

        while (current != null)
        {
            if(current.key.equals(key))
            {
                if(prev == null)
                {
                    ht[index] = current.next;
                }
                else
                {
                    prev.next = current.next;
                }
                return;
            }
            prev = current;
            current = current.next;
        }
    }

public static int diamondCount(Stack stack, String str) {
        int count = 0;

        for(int i = 0; i < str.length(); i++)
        {
            char c = str.charAt(i);
            if (c !='.')
              {
                if (c == '<')
                {
                    stack.push ('<');
                }
                else if(c == '>')
                {
                    if (!stack.isEmpty())
                    { 
                         if(stack.peek().equals('<'))
                         {
                            stack.pop();
                            count++;
                         }
                    }
                }
              }
        }
        return count; 
    }

public static String removeConsecDups(String word){
	LinkedListQueue queue = new LinkedListQueue();

    for (int i = 0; i<word.length(); i++)
    {
        queue.enqueue(word.charAt(i));
    }

    String result = "";
    char prevCH = ' ';



    while(!queue.isEmpty())
    {
        char currCH = (char)queue.dequeue();
    if (currCH != prevCH)
    {
        result += currCH;
        prevCH = currCH;
    }
    }
	return result; //remove this line once you're ready
    }
