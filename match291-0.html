 private int hashFunction( Integer key ){
        if (key == null) return -1;
        return (key + 3) % 6;
}

  public void remove(Integer key){
    int index = hashFunction(key);
    if (index == -1 || ht[index] == null) {


        return;
    }
    PairNode temp = ht[index];
    PairNode prev = null;
    while (temp != null) {
        if (temp.key.equals(key)) {
            if (prev == null) {
                ht[index] = temp.next;
            } else {
                prev.next = temp.next;
            }
            return;
        }
        prev = temp;
        temp = temp.next;
  }
}

  public static int diamondCount(Stack stack, String str) {
      int diamondCount = 0;
      for(int i =0;i< str.length(); i++) {
          char ch = str.charAt(i);
          if (ch == '<') {
              stack.push(ch);
          } else if (ch == '>') {
              if (!stack.isEmpty())  {
                  stack.pop();
                  diamondCount++;
              }
          }
      }
      return diamondCount;
    }

public static String removeConsecDups(String word){
        LinkedListQueue queue = new LinkedListQueue();


        char lastChar = 0; 
        for (int i = 0; i < word.length(); i++) {
            char currentChar = word.charAt(i);
         if (i == 0 || currentChar != lastChar) {
            queue.enqueue(currentChar);
            lastChar = currentChar;
        }
        }
    String result = "";
    while (!queue.isEmpty()) {
    result += queue.dequeue();
    }
    return result;
}
